Jacob's Turtle Exercise

by Jacob Hill

17 Jan 2014

import turtle
wn = turtle.Screen()

wn.bgcolor("lightblue")

sun = turtle.Turtle()
sun.pensize(5)
sun.color("orange")
sun.hideturtle()

grass = turtle.Turtle()
grass.pensize(1)
grass.color("green")
grass.shape()
grass.speed(10)

ray = turtle.Turtle()
ray.pensize(4)
ray.color("yellow")
ray.hideturtle()
ray.speed(10)

ground = turtle.Turtle()
ground.pensize(4)
ground.color("brown")
ground.shape()
ground.speed(10)

leaf = turtle.Turtle()
leaf.pensize(5)
leaf.color("green")
leaf.speed(10)

trunk = turtle.Turtle()
trunk.pensize(3)
trunk.color("brown")
trunk.speed(10)


sun.up()
sun.goto(115, 135)
sun.down()
sun.speed(10)

trunk.up()
trunk.goto(-185, -40)
trunk.down()
trunk.begin_fill()
trunk.right(91)
trunk.goto(-186, -251)
trunk.left(91)
trunk.forward(25)
trunk.left(91)
trunk.goto(-167, -40)
trunk.end_fill()
