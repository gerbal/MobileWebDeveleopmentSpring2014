from test import testEqual

def remove(substr,theStr):
    # your code here
    index =theStr.find(substr);
    if index==-1:
        return theStr
    if index>=0:
        return theStr[:index] + theStr[index+len(substr):]

from test import testEqual
def replace(s, old, new):
    lst = s.split(old) 
    return new.join(lst)

testEqual(replace('Mississippi', 'i', 'I'), 'MIssIssIppI')
